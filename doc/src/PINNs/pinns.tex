\section{Motivation}

%---------------------------------------------------------
\begin{frame}{Quantum Dots as Artificial Atoms}
\begin{itemize}
\item Semiconductor-confined electrons
\item Tunable confinement $\omega$
\item Strong Coulomb correlations
\item Few-body → mesoscopic regime
\end{itemize}

\vspace{0.3cm}
Goal:
\[
\text{Understand correlation-driven structure formation}
\]
\end{frame}

%---------------------------------------------------------
\begin{frame}{The Many-Body Hamiltonian}
\[
H =
\sum_i
\left(
-\frac{1}{2}\nabla_i^2
+
\frac{1}{2}\omega^2 r_i^2
\right)
+
\sum_{i<j}\frac{1}{|r_i-r_j|}
\]

Competition:
\begin{itemize}
\item Kinetic energy
\item Trap confinement
\item Coulomb repulsion
\end{itemize}
\end{frame}

%---------------------------------------------------------
\begin{frame}{Why Is This Hard?}
\begin{itemize}
\item Exponential Hilbert space growth
\item Strong correlation effects
\item Near-Wigner crystallization at weak confinement
\end{itemize}

Traditional methods:
\begin{itemize}
\item Exact diagonalization (small $N$)
\item Diffusion Monte Carlo
\item Configuration interaction
\end{itemize}
\end{frame}

\section{Physics-Informed Neural Networks}

%---------------------------------------------------------
\begin{frame}{Idea of PINNs}
Instead of sampling:
\[
H\Psi = E\Psi
\]

Minimize PDE residual:
\[
\mathcal{L}
=
\langle |(H-E)\Psi_\theta|^2 \rangle
\]

Advantages:
\begin{itemize}
\item No stochastic Monte Carlo noise
\item Direct enforcement of Schrödinger equation
\item Continuous representation
\end{itemize}
\end{frame}

%---------------------------------------------------------
\begin{frame}{Wavefunction Ansatz}
\[
\Psi(R)
=
\text{SD}(R')
\exp(U(R) + W_\theta(R))
\]

Components:
\begin{itemize}
\item Slater determinant
\item Jastrow correlation
\item Neural correlator
\item Optional backflow
\end{itemize}
\end{frame}

%---------------------------------------------------------
\begin{frame}{Neural Correlator}
Permutation invariant construction:

\[
W_\theta(R)
=
\rho\Big(h_\phi \parallel h_\psi \parallel g_1 \parallel g_2\Big)
\]

Features:
\begin{itemize}
\item Pairwise distances
\item Safe-distance channels
\item Radial global scalars
\end{itemize}
\end{frame}

%---------------------------------------------------------
\begin{frame}{Backflow Deformation}
Coordinate transformation:

\[
R' = R + \Delta_\beta(R)
\]

Neural backflow:
\begin{itemize}
\item Message passing
\item $O(N^2)$ scaling
\end{itemize}
\end{frame}

Rank-based shell identification:

Examples:

Collapse under strong confinement:
\[
(1,N-1)

%---------------------------------------------------------
\begin{frame}{Angular Ordering}
Registered bond order:
\[
\psi^{\text{reg}}_{n_s}
=
\left|
\frac{1}{n_s}
\sum_{j\in s} e^{in_s\theta_j^{\text{reg}}}
\right|
\]

Strong ordering at weak $\omega$.
\end{frame}

%---------------------------------------------------------
\begin{frame}{Shell Disorder}
Lindemann-type fluctuation:

\[
L_{s,\text{reg}}
\]

Weak confinement:
\begin{itemize}
\item Small fluctuations
\item Stiff shells
\end{itemize}

Strong confinement:
\begin{itemize}
\item Melting behavior
\end{itemize}
\end{frame}
