% Slide 1: Vision & Strategic context
\begin{frame}{Vision \& Strategic Context}
\begin{itemize}
  \item \textbf{Vision:} Establish Norway as a nationally and internationally leading hub for theoretical advances in quantum state preparation, control, and algorithms for quantum computing and sensing. (focus on state engineering across multiple platforms). 
  \item Interdisciplinary emphasis: condensed-matter / many-body theory, quantum chemistry, quantum algorithms, computational science, ML/AI, and quantum sensing. 
  \item Strategic fit: aligns with national priorities for quantum readiness and leverages existing cloud quantum HW and national HPC.
\item Collaboration UiO, USN, OsloMet, Kristiania University College and Simula Research Laboratory
\end{itemize}
\end{frame}


% Slide 2: Research Objectives (ROs)
\begin{frame}{Core Research Objectives (ROs)}
  \begin{enumerate}
  \item \textbf{RO1 — Many-body physics and entanglement:} develop analytical and numerical tools to characterize multipartite entanglement and its use for state preparation and gates. 
  \item \textbf{RO2 — QML and state engineering:} integrate ML to optimize state preparation, classification and control; study when quantum resources improve learning.  
  \item \textbf{RO3 — Quantum sensing:} design and benchmark protocols and states (squeezed/entangled) that improve metrological precision on realistic platforms.  
  \item \textbf{RO4 — Quantum algorithms and control:} use advanced state preparation to build algorithms and error-correction methods that reduce resource overheads.  
  \end{enumerate}
\end{frame}

% Slide 3: Implementation — Work Packages & Methods
\begin{frame}{Implementation: Work Packages and Methods}
  \textbf{Main Work Packages (WPs):}
  \begin{itemize}
  \item \textbf{WP1:} Entanglement in many-body systems (benchmarks, QMC, CI, CC, time-dependent methods). 
  \item \textbf{WP2:} Quantum machine learning libraries, benchmarks, hybrid workflows.
  \item \textbf{WP3:} Hybrid quantum-state engineering, QC error mitigation and RL-driven protocol optimization.
  \item \textbf{WP4:} Quantum sensing & metrology — sensitivity benchmarks and protocols.
  \item \textbf{WP5:} Quantum algorithms, simulation and error-correction design for NISQ/near-term devices. 
  \end{itemize}

  \bigskip
  \textbf{Methods \& tools:} analytic theory, large-scale numerics (QMC, DFT, CI/CC), ML (classical and QML), prototype circuit design and cloud-accessible quantum simulators/HW.
\end{frame}

% Slide 4: Impact, Management, Resources
\begin{frame}{Impact, Management and Resources}
  \textbf{Anticipated Impact}
  \begin{itemize}
  \item Scientific: new theoretical limits, state engineering protocols, entanglement benchmarks. 
  \item Technological: improved gate fidelities, sensor designs, transferable software and control toolkits. 
  \item Educational / workforce: train PhD/postdocs (target: 7 PhD + 3 postdocs), courses, workshops, open-source code and datasets.  National driver in developing PhD and MSc courses and EVU courses in Quantum Technologies
  \end{itemize}

  \bigskip
  \textbf{Management and Budget}
  \begin{itemize}
  \item Director and WP leads form Executive Committee; external Advisory Board for strategic guidance. Regular milestone reviews and risk mitigation (six-month internal evals). 
  \item Budget (RCN NOK 61M): personnel-dominated ($70\%$), travel/workshops ($20\%$), HPC and ops ($10\%$). Plan to hire 7 PhD and 3 postdocs.  
  \end{itemize}
\end{frame}

% Slide 5: One-slide summary
\begin{frame}{One-slide summary}
  \textbf{QOMPAS in one line:} A theory-centered, interdisciplinary center that unites many-body physics, quantum control, ML and algorithms to deliver state-preparation protocols, sensor designs and algorithmic advances with strong educational and open-science outputs.  
\end{frame}
